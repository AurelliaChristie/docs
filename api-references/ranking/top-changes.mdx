---
title: "Top Company Changes Ranked"
api: "GET /companies/top-changes/"
description: "Return a list of top tickers changes in a given period based on certain classification (top_gainers or top_losers)"
---

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors API key.
</ParamField>

### Query Parameters

<ParamField
  query="classifications"
  type="string"
  default="all"
>
  Classification for ranking the top changes in a given period (default to all).
  <AccordionGroup>
    <Accordion title="Available classifications">
      `top_gainers`, `top_losers`
    </Accordion>
    <Accordion title="Format">
      Use comma to separate each classification when retrieving data of more
      than one classification.
    </Accordion>
    <Accordion title="API call consumption">
      One classification consumes one API call.
    </Accordion>
  </AccordionGroup>
</ParamField>

<ParamField query="n_stock" type="integer" default="5">
  Number of tickers to be retrieved (defaults to 5, with a maximum of 10).
</ParamField>

<ParamField
  query="periods"
  type="string"
  default="all"
>
  Periods for ranking the top changes.
  <AccordionGroup>
    <Accordion title="Available periods">
      `1d`, `7d`, `14d`, `30d`, `365d`
    </Accordion>
    <Accordion title="Format">
      Use comma to separate each period when retrieving data of more
      than one period.
    </Accordion>
    <Accordion title="API call consumption">
      One period consumes one API call.
    </Accordion>
  </AccordionGroup>
</ParamField>

<ParamField query="sub_sector" type="string">
  Subsector from which the list of top company changes is to be retrieved.
  <AccordionGroup>
    <Accordion title="Get the available sub_sector">
      Get the available `sub_sector` from the [Subsectors](./subsectors)
      endpoint.
    </Accordion>
    <Accordion title="Examples of the sub_sector">
      `banks`, `financing-service`, `insurance`, `retailing`, `tobacco`
    </Accordion>
    <Accordion title="Format">
      The API endpoint only accepts `sub_sector` in the **kebab case** format
      (lowercase and separated by "-").
    </Accordion>
  </AccordionGroup>
</ParamField>
